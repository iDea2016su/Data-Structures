class Item
{
public:
	Item();
	~Item();
	Item *next;
	Item *before;
	int value1;
	int value2;
	int value3;
};

class List
{
public:
	List();
	~List();
	Item* head;
	Item* end;
	int len;
	Item Find(Item &item);
};

{
 Â  value1 = -1;
   value2 = -1;
   value3 = -1;

}

Item List::Find(Item & item)
{
	Item * tmp = head;
	for (int i = 0; i < len; i++)
	{
		tmp = tmp->next;
		if ((tmp->value1 == item.value1) || (tmp->value2 == item.value2) || (tmp->value3 == item.value3))
		{
			return *tmp;
		}
	}
	return;
}
