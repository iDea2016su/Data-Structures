#include <iostream>
#include <string>

using namespace std;
int Check(string & s)
{
  int left = 0;
  int right = 0;
  int len = s.length();
  for(int i=0;i<len;i++)
  {
     if(s[i]=='(')
     {
	left++;
     }
     else if(s[i]==')')
     {
	right++;
     }
     if(right>left)
     {
	return -1;
     }
  }
  if(right == left)
  {
    return 1;
  }
  return -1;
}

int main()
{
   cout<<"hello"<<endl;
   string str = "jwhefo(lnkn))nkjnl()()(())nkj";
   cout<<str<<endl;
   if(Check(str)>0)
   { 
     cout<<"good string"<<endl;
   }
   else
   {
     cout<<"bad string"<<endl;
   }
   return 0;
}
